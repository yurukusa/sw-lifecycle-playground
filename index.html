Basic Registration

// Register a Service Worker if ('serviceWorker' in navigator) { const registration = await navigator.serviceWorker.register('/sw.js', { scope: '/' }); console.log('SW registered:', registration.scope); // Monitor state changes registration.addEventListener('updatefound', () => { const newSW = registration.installing; console.log('New SW found, state:', newSW.state); newSW.addEventListener('statechange', () => { console.log('SW state changed:', newSW.state); // "installing" → "installed" → "activating" → "activated" }); }); }

Service Worker File (sw.js)

const CACHE_NAME = 'v1'; const ASSETS = ['/', '/index.html', '/style.css', '/app.js']; // Install: cache assets self.addEventListener('install', (event) => { event.waitUntil( caches.open(CACHE_NAME).then((cache) => cache.addAll(ASSETS)) ); // Optional: skip waiting to activate immediately // self.skipWaiting(); }); // Activate: clean old caches self.addEventListener('activate', (event) => { event.waitUntil( caches.keys().then((names) => Promise.all( names.filter(n => n !== CACHE_NAME).map(n => caches.delete(n)) ) ) ); // Optional: take control of all clients immediately // self.clients.claim(); }); // Fetch: serve from cache, fallback to network self.addEventListener('fetch', (event) => { event.respondWith( caches.match(event.request).then((cached) => cached || fetch(event.request)) ); });

Update Flow

// Force update check const registration = await navigator.serviceWorker.getRegistration(); await registration.update(); // Listen for controller change (new SW took over) navigator.serviceWorker.addEventListener('controllerchange', () => { console.log('New Service Worker is now controlling the page'); // Optionally reload to ensure fresh content // window.location.reload(); }); // In SW: force activation (skip waiting phase) // sw.js self.addEventListener('install', () => { self.skipWaiting(); // Don't wait for old SW to be replaced }); self.addEventListener('activate', (event) => { event.waitUntil(self.clients.claim()); // Take control immediately });

Communication (postMessage)

// Page → Service Worker navigator.serviceWorker.controller.postMessage({ type: 'CACHE_URLS', urls: ['/api/data'] }); // Service Worker → Page // In sw.js: self.addEventListener('message', (event) => { if (event.data.type === 'CACHE_URLS') { caches.open('dynamic').then(cache => cache.addAll(event.data.urls)); // Reply to specific client event.source.postMessage({ type: 'CACHED', count: event.data.urls.length }); } }); // Listen for messages from SW (in page): navigator.serviceWorker.addEventListener('message', (event) => { console.log('From SW:', event.data); });
